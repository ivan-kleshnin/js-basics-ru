# 2.4 Основные инструкции

```js
let xs = [1, 1, 1]

// Цикл while
let i = 0
while (i <= xs.length) {
  console.log(xs[i])
  i++
}

// Цикл for
for (let i = 0; i < xs.lenght; i++) {
  console.log(xs[i])
}

// Цикл for-of
for (let x of [1, 1, 1]) {
  // ...
}

// return
function head(xs) {
  return xs[0]
}

// Перехват ошибок
try {                 // Только синхронные ошибки
  throw Error("oops") // Выброс ошибки вручную
} catch (e) {         // Нет возможности перехвата по типу
  console.log(e)
}
                      // Нет поддержки finally / else веток
```
