# 2.4 Основные инструкции

```js
// Условие if-else
if (user.role == "admin") {
  // ...
} else if (user.role == "manager") {
  // ...
} else {
  // ...
}

// Условие switch-case
switch (user.role) {
  case "admin":
    // ...
    break
  case "manager":
    // ...
    break
  default:
    // ...
}

// Цикл while
let xs = [1, 1, 1]
let i = 0
while (i <= xs.length) {
  console.log(xs[i])
  i++
}

// Цикл for
let xs = [1, 1, 1]
for (let i = 0; i < xs.length; i++) {
  console.log(xs[i])
}

// Цикл for-of
let xs = [1, 1, 1]
for (let x of xs) {
  // ...
}

// Возврат значения
function head(xs) {
  return xs[0]
}

// Перехват ошибок
try {                 // Только синхронные ошибки
  throw Error("oops") // Выброс ошибки вручную
} catch (e) {         // Нет возможности перехвата по типу
  console.log(e)
}
                      // Нет поддержки finally / else веток
```
